ACTOR CacoBrute
{
  Health 650
  Radius 31
  Height 56
  Mass 400
  Speed 8
  PainChance 128
  Monster
  +FLOAT
  +NOGRAVITY
  +missilemore
  +missileevenmore
  minmissilechance 2
  SeeSound "caco/sight"
  PainSound "caco/pain"
  DeathSound "caco/death"
  ActiveSound "caco/active"
  Obituary "%o is now the Cacodemon Brute's lunch."
  States
  {
  Spawn:
    SPIK A 10 A_Look
    Loop
  See:
    SPIK A 3 A_Chase
    Loop
  Missile:
    SPIK BC 5 A_FaceTarget
    SPIK D 0 Bright A_custommissile("cacodemonball",24,-8,0)
    SPIK D 5 Bright A_custommissile("cacodemonball",24,8,0)
    SPIK C 3 A_FaceTarget
    SPIK D 0 Bright A_custommissile("cacodemonball",24,-8,0)
    SPIK D 5 Bright A_custommissile("cacodemonball",24,8,0)
    Goto See
  Pain:
    SPIK B 3
    SPIK B 3 A_Pain
    SPIK B 6
    Goto See
  Death:
    SPIK E 8
    SPIK F 8 A_Scream
    SPIK GHIJ 8
    SPIK K 8 A_NoBlocking
    SPIK L -1 A_SetFloorClip
    Stop
 
  }
}
